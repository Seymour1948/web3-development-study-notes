Ethers JS Notes

installation

npm install --save ethers in a empty directory after "npm init -y"

require

const { ethers } = require("ethers"); import library

provider

const provider = new ethers.providers.JsonRpcProvider("HTTP://0.0.0.0:7545") string argument for ganache

get address

const signer = await provider.getSigner(); obtain first account

transfer

signer.transfer({to: "0xdsds...", value: ethers.utils.parseEther("1.0")}); send ether from signer account

call

provider.call(from: signer, to: "0x989...", data: ethers.utils.kek256(ethers.utils.toUtf8Bytes("myUint()"))) Low level call

ethers.utils.kek256(ethers.utils.toUtf8Bytes("myUint()")) This part hash our string
